查找最接近的元素

查找最接近的元素

90分代码

#include<cstdio>
#include<algorithm>
#include<cstring> 
#include<queue>
using namespace std;
int n,s[100002],m,x;
inline int find(int xx,int l,int r)
{
    if(xx<s[l])return l-1;
    if(l<r)
    {
        int mid = (l+r)>>1;
        if(xx==s[mid])
        return mid;
        if(xx<s[mid])return find(xx,l,mid);
        return find(xx,mid+1,r);
    }
    return r-1;
}
inline int search(int xx)
{
    if(xx<=s[1])return s[1];
    if(xx>=s[n])return s[n];
    int tmp = find(xx,1,n);
    //find用来寻找等于x或者比x小的第一个值
    if(xx==s[tmp])return s[tmp];
    if(xx-s[tmp]<=s[tmp+1]-xx&&tmp!=0)
        return s[tmp];
    return s[tmp+1];
}
int main()
{
    scanf("%d",&n);
    for(int i=1;i<=n;i++)
    {
        scanf("%d",&s[i]);
        if(s[i]==s[i-1])i--,n--;
    }
    s[0]=-1;
    scanf("%d",&m);
    while(m--)
    {
        scanf("%d",&x);
        printf("%d\n",search(x));
    }
    return 0;
}
